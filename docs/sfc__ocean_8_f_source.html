76! q1 - real, surface layer mean specific humidity im !

80! lseaspray- logical, .t. for parameterization for sea spray 1 !
81! fm - real, a stability profile function for momentum im !
82! fm10 - real, a stability profile function for momentum im !
83! at 10m !

85! prslki - real, im !

87! wind - real, wind speed (m/s) im !
88! flag_iter- logical, im !

92! !

94! qsurf - real, specific humidity at sfc im !
95! cmm - real, im !
96! chh - real, im !
97! gflux - real, ground heat flux (zero for ocean) im !

99! hflx - real, sensible heat flux im !

111 & zero = 0.0_kind_phys, qmin = 1.0e-8_kind_phys
112! --- inputs:
113 integer, intent(in) :: im
114 real (kind=kind_phys), intent(in) :: hvap, cp, rd, &

124 logical, dimension(:), intent(in) :: flag_iter, wet

128 logical, intent(in) :: use_med_flux
129

153!

156!
157! real (kind=kind_phys), parameter :: alps=0.5, bets=0.5, gams=0.1,
158! real (kind=kind_phys), parameter :: alps=0.5, bets=0.5, gams=0.0,
159! real (kind=kind_phys), parameter :: alps=1.0, bets=1.0, gams=0.2,

161 & ws10cr=30., conlf=7.2e-9, consf=6.4e-8
162!
163!======================================================================================================
164!===> ... begin here
165!
166! -- ... initialize CCPP error handling variables
167 errmsg = ''
168 errflg = 0
169
170 cpinv = one/cp
171 hvapi = one/hvap
172 elocp = hvap/cp
173!
174!> - Flag for open water
175 do i = 1, im
176 flag(i) = (wet(i) .and. flag_iter(i) .and. use_lake_model(i)/=1)
177!> - Initialize variables. all units are supposedly m.k.s. unless specified
178! ps is in pascals, wind is wind speed,
179! rho is density, qss is sat. hum. at surface
180
181 if ( flag(i) ) then
182 if (use_med_flux) then
183 q0(i) = max( q1(i), qmin )
184 rho(i) = prsl1(i) / (rd*t1(i)*(one + rvrdm1*q0(i)))
185
186 if (use_oceanuv) then
187 windrel(i)=sqrt( (u1(i)-usfco(i))**2+(v1(i)-vsfco(i))**2 )
188 tem = ch(i) * windrel(i)
189 cmm(i) = cm(i) * windrel(i)
190 else
191 tem = ch(i) * wind(i)
192 cmm(i) = cm(i) * wind(i)
193 endif
194
195 chh(i) = rho(i) * tem
196
197 hflx(i) = dtsfc_med(i)
198 evap(i) = dqsfc_med(i)
199
200 qsurf(i) = q1(i) + dqsfc_med(i) / (hvap*chh(i))
201 gflux(i) = zero
202 else
203 q0(i) = max( q1(i), qmin )
204 rho(i) = prsl1(i) / (rd*t1(i)*(one + rvrdm1*q0(i)))
205
206 qss = fpvs( tskin(i) )
207 qss = eps*qss / (ps(i) + epsm1*qss)
208
209! --- ... rcp = rho cp ch v
210
211 if (use_oceanuv) then
212 windrel(i)=sqrt( (u1(i)-usfco(i))**2+(v1(i)-vsfco(i))**2 )
213 rch = rho(i) * cp * ch(i) * windrel(i)
214 tem = ch(i) * windrel(i)
215 cmm(i) = cm(i) * windrel(i)
216 else
217 rch = rho(i) * cp * ch(i) * wind(i)
218 tem = ch(i) * wind(i)
219 cmm(i) = cm(i) * wind(i)
220 endif
221
222 chh(i) = rho(i) * tem
223
224!> - Calcualte sensible and latent heat flux over open water
225
226 hflx(i) = rch * (tskin(i) - t1(i) * prslki(i))
227
228 evap(i) = elocp * rch * (qss - q0(i))
229
230 qsurf(i) = qss
231 gflux(i) = zero
232 endif
233 endif
234 enddo
235!
236!> - Include sea spray effects
237!
238 do i=1,im
239 if(lseaspray .and. flag(i)) then
240 f10m = fm10(i) / fm(i)
241 u10m = f10m * u1(i)
242 v10m = f10m * v1(i)
243 ws10 = sqrt(u10m*u10m + v10m*v10m)
244 ws10 = max(ws10,1.)
245 ws10 = min(ws10,ws10cr)
246 tem = .015 * ws10 * ws10
247 ru10 = 1. - .087 * log(10./tem)
248 qss1 = fpvs(t1(i))
249 qss1 = eps * qss1 / (prsl1(i) + epsm1 * qss1)
250 tem = rd * cp * t1(i) * t1(i)
251 tem = 1. + eps * hvap * hvap * qss1 / tem
252 bb1 = 1. / tem
253 evaps = conlf * (ws10**5.4) * ru10 * bb1
254 evaps = evaps * rho(i) * hvap * (qss1 - q0(i))
255 evap(i) = evap(i) + alps * evaps
256 hflxs = consf * (ws10**3.4) * ru10
257 hflxs = hflxs * rho(i) * cp * (tskin(i) - t1(i))
258 ptem = alps - gams
259 hflx(i) = hflx(i) + bets * hflxs - ptem * evaps
260 endif
261 enddo
262!
263 do i = 1, im
264 if ( flag(i) ) then
265 tem = 1.0 / rho(i)
266 hflx(i) = hflx(i) * tem * cpinv
267 evap(i) = evap(i) * tem * hvapi
268 ep(i) = evap(i)
269 endif
270 enddo
271!
272 return

274 end subroutine sfc_ocean_run
