7!> @brief Land IAU (Incremental Analysis Update) module,
8!> for the NoahMP soil/snow temperature (can be extended to include soil moisture)
9
10!! \section land_iau_mod
11!> - reads settings from namelist file (which indicates if IAU increments are available or not)
12!> - reads in DA increments from GSI/JEDI DA at the start of (the DA) cycle
13!> - maps increments to FV3 grid points belonging to mpi process
14!> - interpolates temporally (with filter-weights if required by configuration)
15!> - updates states with the interpolated increments
16
17!> March, 2024: Tseganeh Z. Gichamo, (EMC) based on the FV3 IAU mod
18!> by Xi.Chen <xi.chen@noaa.gov> and Philip Pegion, PSL <philip.pegion@noaa.gov>
19!-------------------------------------------------------------------------------

229subroutine land_iau_mod_init (Land_IAU_Control, Land_IAU_Data, Land_IAU_State, errmsg, errflg)
230 type (land_iau_control_type), intent(inout) :: land_iau_control
231 type (land_iau_external_data_type), intent(inout) :: land_iau_data
232 type(land_iau_state_type), intent(inout) :: land_iau_state
233 character(len=*), intent( out) :: errmsg
234 integer, intent( out) :: errflg
235
236 ! local
237 character(len=128) :: fname
238 real(kind=kind_phys) :: sx, wx, wt, normfact, dtp
239 integer :: k, nstep, kstep
240 integer :: nfilesall, ntimesall
241 integer, allocatable :: idt(:)
242 integer :: nlon, nlat
243 logical :: exists
244 integer :: ncid, dimid, varid, status, idim
245
246 real(kind=kind_phys) :: dt !, rdt
247 integer :: im, jm, km, nfiles, ntimes
248
249 integer :: is, ie, js, je
250 integer :: npz
251 integer :: i, j
252
253 !Errors messages handled through CCPP error handling variables
254 errmsg = ''
255 errflg = 0
256
257 npz = land_iau_control%lsoil
258 km = land_iau_control%lsoil
259
260 is = land_iau_control%isc
261 ie = is + land_iau_control%nx-1
262 js = land_iau_control%jsc
263 je = js + land_iau_control%ny-1
264 nlon = land_iau_control%nx
265 nlat = land_iau_control%ny
266
267 ! allocate arrays that will hold iau state
268 allocate(land_iau_data%stc_inc(nlon, nlat, km))
269 allocate(land_iau_data%slc_inc(nlon, nlat, km))
270
271 land_iau_data%hr1=land_iau_control%iaufhrs(1)
272 land_iau_data%wt = 1.0 ! IAU increment filter weights (default 1.0)
273 land_iau_data%wt_normfact = 1.0
274 if (land_iau_control%iau_filter_increments) then
275 ! compute increment filter weights, sum to obtain normalization factor
276 dtp=land_iau_control%dtp
277 nstep = 0.5*land_iau_control%iau_delthrs*3600/dtp
278 ! compute normalization factor for filter weights
279 normfact = 0.
280 do k=1,2*nstep+1
281 kstep = k-1-nstep
282 sx = acos(-1.)*kstep/nstep
283 wx = acos(-1.)*kstep/(nstep+1)
284 if (kstep .ne. 0) then
285 wt = sin(wx)/wx*sin(sx)/sx
286 else
287 wt = 1.0
288 endif
289 normfact = normfact + wt
290 if (land_iau_control%me == land_iau_control%mpi_root) then
291 print *,'Land IAU init: IAU filter weights params k, kstep, wt ',k, kstep, wt
292 endif
293 enddo
294 land_iau_data%wt_normfact = (2*nstep+1)/normfact
295 endif
296
297 ! increment files are in fv3 tiles
298 if (trim(land_iau_control%iau_inc_files(1)) .eq. '' .or. land_iau_control%iaufhrs(1) .lt. 0) then ! only 1 file expected
299 errmsg = "Error! in Land IAU init: increment file name is empty or iaufhrs(1) is negative"
300 errflg = 1
301 return
302 endif
303 if (land_iau_control%me == land_iau_control%mpi_root) then
304 print*,"Land_iau_init: Increment file name: ", trim(adjustl(land_iau_control%iau_inc_files(1)))
305 endif
306
307 ! determine number of valid forecast hours; read from the increment file ("Time" dim)
308 if (land_iau_control%me == land_iau_control%mpi_root) then
309 print *, "Land_iau_init: timesetps and forecast times (in hours) with valid increment values"
310 endif
311 ntimesall = size(land_iau_control%iaufhrs)
312 ntimes = 0
313 do k=1,ntimesall
314 if (land_iau_control%iaufhrs(k) .lt. 0) exit
315 if (land_iau_control%me == land_iau_control%mpi_root) then
316 print *,k, " fhour ", land_iau_control%iaufhrs(k)
317 endif
318 ntimes = ntimes + 1
319 enddo
320
321 land_iau_control%ntimes = ntimes
322 if (ntimes < 1) then
323 errmsg = "Error! in Land IAU init: ntimes < 1 (no valid hour with increments); do_land_iau should not be .true."
324 errflg = 1
325 return
326 endif
327 if (ntimes > 1) then
328 allocate(idt(ntimes-1))
329 idt = land_iau_control%iaufhrs(2:ntimes)-land_iau_control%iaufhrs(1:ntimes-1)
330 do k=1,ntimes-1
331 if (idt(k) .ne. land_iau_control%iaufhrs(2)-land_iau_control%iaufhrs(1)) then
332 errmsg = 'Fatal error in land_iau_init. forecast intervals in iaufhrs must be constant'
333 errflg = 1
334 return
335 endif
336 enddo
337 deallocate(idt)
338 endif
339 dt = (land_iau_control%iau_delthrs*3600.)
340 land_iau_data%rdt = 1.0/dt !rdt
341
342 ! Read all increment files at iau init time (at beginning of cycle)
343 ! increments are already in the fv3 grid--no need for interpolation
344 call read_iau_forcing_fv3(land_iau_control, land_iau_state%stc_inc, land_iau_state%slc_inc, errmsg, errflg)
345 if (errflg .ne. 0) return
346
347 if (ntimes.EQ.1) then ! only need to get incrments once since constant forcing over window
348 call setiauforcing(land_iau_control, land_iau_data, land_iau_state)
349 land_iau_data%itnext = 0
350 endif
351 if (ntimes.GT.1) then !have increments at multiple forecast hours,
352 ! but only need 2 at a time and interpoalte for timesteps between them
353 ! interpolation is done in land_iau_mod_getiauforcing
354 land_iau_data%hr2=land_iau_control%iaufhrs(2)
355 land_iau_data%itnext = 2
356 endif
357
358end subroutine land_iau_mod_init

382 subroutine land_iau_mod_getiauforcing(Land_IAU_Control, Land_IAU_Data, Land_IAU_State, errmsg, errflg)
383
384 implicit none
385 type(land_iau_control_type), intent(inout) :: land_iau_control
386 type(land_iau_external_data_type), intent(inout) :: land_iau_data
387 type(land_iau_state_type), intent(in) :: land_iau_state
388 character(len=*), intent(out) :: errmsg
389 integer, intent(out) :: errflg
390 real(kind=kind_phys) t1,t2,sx,wx,wt,dtp
391 integer n,i,j,k,kstep,nstep
392 integer :: ntimes
393
394 ! Initialize CCPP error handling variables
395 errmsg = ''
396 errflg = 0
397
398 ntimes = land_iau_control%ntimes
399
400 land_iau_data%in_interval=.false.
401 if (ntimes.LE.0) then
402 errmsg = 'called land_iau_mod_getiauforcing, but ntimes <=0, probably there is no increment file. Exiting.'
403 errflg = 1
404 return
405 endif
406
407 if (ntimes .eq. 1) then
408 t1 = land_iau_control%iaufhrs(1)-0.5*land_iau_control%iau_delthrs
409 t2 = land_iau_control%iaufhrs(1)+0.5*land_iau_control%iau_delthrs
410 else
411 t1 = land_iau_control%iaufhrs(1)
412 t2 = land_iau_control%iaufhrs(ntimes)
413 endif
414 if (land_iau_control%iau_filter_increments) then
415 ! compute increment filter weight
416 ! t1 is beginning of window, t2 end of window, and Land_IAU_Control%fhour is current time
417 ! in window kstep=-nstep,nstep (2*nstep+1 total) with time step of Land_IAU_Control%dtp
418 dtp=land_iau_control%dtp
419 nstep = 0.5*land_iau_control%iau_delthrs*3600/dtp
420 ! compute normalized filter weight
421 kstep = ((land_iau_control%fhour-t1) - 0.5*land_iau_control%iau_delthrs)*3600./dtp
422 if (land_iau_control%fhour >= t1 .and. land_iau_control%fhour < t2) then
423 sx = acos(-1.)*kstep/nstep
424 wx = acos(-1.)*kstep/(nstep+1)
425 if (kstep .ne. 0) then
426 wt = (sin(wx)/wx*sin(sx)/sx)
427 else
428 wt = 1.
429 endif
430 land_iau_data%wt = land_iau_data%wt_normfact*wt
431 else
432 land_iau_data%wt = 0.
433 endif
434 endif
435
436 if (ntimes.EQ.1) then
437 ! check to see if we are in the IAU window, no need to update the states since they are fixed over the window
438 if ( land_iau_control%fhour <= t1 .or. land_iau_control%fhour > t2 ) then
439 land_iau_data%in_interval=.false.
440 else
441 land_iau_data%in_interval=.true.
442 if (land_iau_control%me == land_iau_control%mpi_root) then
443 print *,'land_iau_mod_getiauforcing: applying forcing at t for t1,t,t2,filter wt rdt ', &
444 t1,land_iau_control%fhour,t2,land_iau_data%wt/land_iau_data%wt_normfact,land_iau_data%rdt
445 endif
446 if (land_iau_control%iau_filter_increments) call setiauforcing(land_iau_control, land_iau_data, land_iau_state)
447 endif
448 return
449 endif
450
451 if (ntimes > 1) then
452 if ( land_iau_control%fhour <= t1 .or. land_iau_control%fhour > t2 ) then
453 land_iau_data%in_interval=.false.
454 else
455 land_iau_data%in_interval=.true.
456 if (land_iau_control%me == land_iau_control%mpi_root) then
457 print *,'land_iau_mod_getiauforcing: applying forcing at t for t1,t,t2,filter wt rdt ', &
458 t1,land_iau_control%fhour,t2,land_iau_data%wt/land_iau_data%wt_normfact,land_iau_data%rdt
459 endif
460 if (land_iau_control%fhour > land_iau_data%hr2) then ! need to read in next increment file
461 land_iau_data%itnext = land_iau_data%itnext + 1
462 land_iau_data%hr1=land_iau_data%hr2
463 land_iau_data%hr2=land_iau_control%iaufhrs(land_iau_data%itnext)
464 endif
465
466 call updateiauforcing(land_iau_control, land_iau_data, land_iau_state)
467 endif
468 endif
469
470 end subroutine land_iau_mod_getiauforcing

472subroutine updateiauforcing(Land_IAU_Control, Land_IAU_Data, Land_IAU_State)
473
474 implicit none
475
476 type (land_iau_control_type), intent(in) :: Land_IAU_Control
477 type(land_iau_external_data_type), intent(inout) :: Land_IAU_Data
478 type(land_iau_state_type), intent(in) :: Land_IAU_State
479 real(kind=kind_phys) delt_t
480 integer i,j,k
481 integer :: is, ie, js, je, npz, t1, t2
482
483 t2 = land_iau_data%itnext
484 t1 = t2 - 1
485 is = 1 ! Land_IAU_Control%isc
486 ie = is + land_iau_control%nx-1
487 js = 1 ! Land_IAU_Control%jsc
488 je = js + land_iau_control%ny-1
489 npz = land_iau_control%lsoil
490
491 delt_t = (land_iau_data%hr2-(land_iau_control%fhour))/(land_iau_data%hr2-land_iau_data%hr1)
492
493 do j = js,je
494 do i = is,ie
495 do k = 1,npz ! do k = 1,n_soill !
496 land_iau_data%stc_inc(i,j,k) =(delt_t*land_iau_state%stc_inc(t1,i,j,k) + (1.-delt_t)* land_iau_state%stc_inc(t2,i,j,k))*land_iau_data%rdt*land_iau_data%wt
497 land_iau_data%slc_inc(i,j,k) =(delt_t*land_iau_state%slc_inc(t1,i,j,k) + (1.-delt_t)* land_iau_state%slc_inc(t2,i,j,k))*land_iau_data%rdt*land_iau_data%wt
498 end do
499 enddo
500 enddo
501 end subroutine updateiauforcing

503 subroutine setiauforcing(Land_IAU_Control, Land_IAU_Data, Land_IAU_State)
504
505 implicit none
