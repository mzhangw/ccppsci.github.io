617 fint*(absliq1(index+1,ib)-absliq1(index,ib)) ))
618 enddo
619 endif

625 else

627 if (ilwcice .eq. 1) then

629 do ib=1,nbandslw

632 enddo
633

635 ! and ebert and curry approach for ice eff radius greater than 131.0 microns.
636 ! no smoothing between the transition of the two methods
637 elseif (ilwcice .eq. 2) then
638 factor = (cld_ref_ice(ij,ik) - 2.) / 3.
639 index = max( 1, min( 42, int( factor ) ))
640 fint = factor - float(index)
641 do ib = 1, nbandslw
642 tau_ice(ib) = max(0., cld_iwp(ij,ik)*(absice2(index,ib) + &
643 fint*(absice2(index+1,ib) - absice2(index,ib)) ))
644 enddo
645 ! 3) Fu's approach for ice effective radius between 4.8 and 135 microns
646 ! (generalized effective size from 5 to 140 microns)
647 elseif (ilwcice .eq. 3) then
648 cld_ref_icetemp = max(5., 1.0315*cld_ref_ice(ij,ik)) ! v4.71 value
649 factor = (cld_ref_icetemp - 2.) / 3.
650 index = max( 1, min( 45, int( factor ) ))
651 fint = factor - float(index)
652 do ib = 1, nbandslw
653 tau_ice(ib) = max(0., cld_iwp(ij,ik)*(absice3(index,ib) + &
654 fint*(absice3(index+1,ib) - absice3(index,ib)) ))
655 enddo
656 endif
657 endif
658 else
659 tau_rain = 0.
660 tau_snow = 0.
661 tau_liq(:) = 0.
662 tau_ice(:) = 0.
663 endif
664 ! Cloud optical depth
665 do ib = 1, nbandslw
666 tau_cld(ij,ik,ib) = tau_ice(ib) + tau_liq(ib)
667 tau_precip(ij,ik,ib) = tau_rain + tau_snow
668 enddo
669 end do
670 end do
671 endif
